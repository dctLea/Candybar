// Fonction pour incrémenter le compteur en cliquant sur le bouton function incrementerCompteur(id) { var compteurElement = document.getElementById(id); var compteurValeur = parseInt(compteurElement.innerText); compteurElement.innerText = compteurValeur + 1; } // Fonction pour incrémenter le compteur manuellement function incrementerCompteurManuel(id) { var compteurElement = document.getElementById(id); var valeurInput = document.getElementById('valeur' + id.substr(-1)).value; var compteurValeur = parseInt(compteurElement.innerText); compteurElement.innerText = compteurValeur + parseInt(valeurInput); }
